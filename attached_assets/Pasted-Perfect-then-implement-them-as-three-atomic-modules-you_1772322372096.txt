A) Core rendering helper

You likely already have some version of this. The important part: render a page to an offscreen canvas at arbitrary size with either “contain” or “cover” crop.

async function renderPageToCanvas(
  page: CanvasPage,
  opts: { outW: number; outH: number; mode: "contain"|"cover"; background?: string },
  drawPage: (ctx: CanvasRenderingContext2D, scale: number, offsetX: number, offsetY: number) => Promise<void>
): Promise<HTMLCanvasElement> {
  const c = document.createElement("canvas");
  c.width = opts.outW;
  c.height = opts.outH;
  const ctx = c.getContext("2d")!;

  // background
  ctx.fillStyle = opts.background ?? "#ffffff";
  ctx.fillRect(0, 0, c.width, c.height);

  const sx = opts.outW / page.width;
  const sy = opts.outH / page.height;
  const scale = opts.mode === "contain" ? Math.min(sx, sy) : Math.max(sx, sy);

  const drawW = page.width * scale;
  const drawH = page.height * scale;
  const offsetX = Math.round((opts.outW - drawW) / 2);
  const offsetY = Math.round((opts.outH - drawH) / 2);

  await drawPage(ctx, scale, offsetX, offsetY);
  return c;
}

async function canvasToAsset(canvas: HTMLCanvasElement, name: string, mime: MediaAsset["mime"] = "image/png"): Promise<MediaAsset> {
  const blob: Blob = await new Promise((resolve) => canvas.toBlob((b) => resolve(b!), mime, 0.92));
  return { name, mime, blob, width: canvas.width, height: canvas.height };
}
B) Grid collage builder (preview 6 pages in one image)
async function buildPreviewGrid(
  pages: CanvasPage[],
  opts: { outW: number; outH: number; cols: number; rows: number; gap: number; pad: number; bg?: string },
  renderThumb: (page: CanvasPage, w: number, h: number) => Promise<HTMLCanvasElement>
): Promise<HTMLCanvasElement> {
  const c = document.createElement("canvas");
  c.width = opts.outW; c.height = opts.outH;
  const ctx = c.getContext("2d")!;
  ctx.fillStyle = opts.bg ?? "#ffffff";
  ctx.fillRect(0, 0, c.width, c.height);

  const cellW = Math.floor((opts.outW - opts.pad*2 - opts.gap*(opts.cols-1)) / opts.cols);
  const cellH = Math.floor((opts.outH - opts.pad*2 - opts.gap*(opts.rows-1)) / opts.rows);

  const max = opts.cols * opts.rows;
  const slice = pages.slice(0, max);

  for (let i = 0; i < slice.length; i++) {
    const col = i % opts.cols;
    const row = Math.floor(i / opts.cols);
    const x = opts.pad + col*(cellW + opts.gap);
    const y = opts.pad + row*(cellH + opts.gap);

    const thumb = await renderThumb(slice[i], cellW, cellH);
    ctx.drawImage(thumb, x, y, cellW, cellH);
  }
  return c;
}

async function generateListingMediaPack(
  pages: CanvasPage[],
  opts: {
    coverPageIndex?: number;
    drawPage: (ctx: CanvasRenderingContext2D, page: CanvasPage, scale: number, ox: number, oy: number) => Promise<void>;
    brandBg?: string;
  }
): Promise<MediaPack> {
  const coverIdx = opts.coverPageIndex ?? 0;
  const cover = pages[coverIdx] ?? pages[0];

  const assets: MediaAsset[] = [];

  // Etsy/Shopify style squares
  {
    const sq = await renderPageToCanvas(
      cover, { outW: 2000, outH: 2000, mode: "cover", background: opts.brandBg },
      (ctx, scale, ox, oy) => opts.drawPage(ctx, cover, scale, ox, oy)
    );
    assets.push(await canvasToAsset(sq, "cover-square-2000.png"));
  }

  // Social share
  {
    const og = await renderPageToCanvas(
      cover, { outW: 1200, outH: 630, mode: "cover", background: opts.brandBg },
      (ctx, scale, ox, oy) => opts.drawPage(ctx, cover, scale, ox, oy)
    );
    assets.push(await canvasToAsset(og, "og-1200x630.png"));
  }

  // IG carousel vertical
  {
    const ig = await renderPageToCanvas(
      cover, { outW: 1080, outH: 1350, mode: "cover", background: opts.brandBg },
      (ctx, scale, ox, oy) => opts.drawPage(ctx, cover, scale, ox, oy)
    );
    assets.push(await canvasToAsset(ig, "ig-1080x1350.png"));
  }

  // Preview grid of first 6 pages
  {
    const grid = await buildPreviewGrid(
      pages, { outW: 2000, outH: 2000, cols: 3, rows: 2, gap: 24, pad: 32, bg: opts.brandBg },
      async (p, w, h) =>
        renderPageToCanvas(p, { outW: w, outH: h, mode: "contain", background: opts.brandBg },
          (ctx, scale, ox, oy) => opts.drawPage(ctx, p, scale, ox, oy))
    );
    assets.push(await canvasToAsset(grid, "preview-grid-6up-2000.png"));
  }

  // (Optional) mockups if you already have frames
  // assets.push(... await generateMockupsFromCover(cover))

  return { assets };
}

2) Canvas → QBank + Flashcards Generator (deterministic → AI → bundle)
A) Deterministic extraction (no prompt guessing)

Extract only “content” elements; ignore logos/footers via tags.

function extractStudyTextFromCanvas(pages: CanvasPage[]): string {
  const lines: string[] = [];

  for (const [pi, page] of pages.entries()) {
    const texts = page.elements
      .filter(e => e.type === "text")
      .filter(e => !["logo","footer","page-number","watermark"].includes((e as any).tag ?? ""))
      .map(e => e as Extract<CanvasElement, {type:"text"}>)
      .sort((a,b) => (a.y - b.y) || (a.x - b.x));

    if (!texts.length) continue;

    lines.push(`PAGE ${pi+1}`);
    for (const t of texts) {
      const clean = (t.text ?? "").trim();
      if (clean) lines.push(clean);
    }
    lines.push(""); // spacer
  }

  return lines.join("\n");
}

B) Overflow scan (fast)

If you already have overflow detection for your existing “Overflow” button, reuse it. Otherwise:

type GateIssue = { severity: "error"|"warn"; code: string; message: string; pageId?: string; elementId?: string };

function scanOverflow(pages: CanvasPage[], measureText: (t: Extract<CanvasElement,{type:"text"}>) => { overflow: boolean }): GateIssue[] {
  const issues: GateIssue[] = [];
  for (const page of pages) {
    for (const el of page.elements) {
      if (el.type !== "text") continue;
      if (["footer","page-number","logo"].includes(el.tag ?? "")) continue;

      const r = measureText(el);
      if (r.overflow) {
        issues.push({
          severity: "error",
          code: "OVERFLOW_TEXT",
          message: "Text overflows its box. Autofit, expand, or paginate before publishing.",
          pageId: page.id,
          elementId: el.id
        });
      }
    }
  }
  return issues;
}
C) Brand enforcement scan
function scanBrandCompliance(pages: CanvasPage[], brand: BrandKit): GateIssue[] {
  const issues: GateIssue[] = [];

  const allowedFonts = new Set([brand.fonts.heading, brand.fonts.body, brand.fonts.mono].filter(Boolean) as string[]);
  const allowedColors = new Set(Object.values(brand.colors));

  for (const page of pages) {
    for (const el of page.elements) {
      if (el.type === "text") {
        if (!allowedFonts.has(el.fontFamily)) {
          issues.push({ severity: "warn", code: "FONT_OFF_BRAND", message: "Non-brand font detected.", pageId: page.id, elementId: el.id });
        }
        if (el.color && !allowedColors.has(el.color)) {
          issues.push({ severity: "warn", code: "COLOR_OFF_BRAND", message: "Non-brand text color detected.", pageId: page.id, elementId: el.id });
        }
      }
      if (el.type === "shape") {
        if (el.fill && !allowedColors.has(el.fill)) {
          issues.push({ severity: "warn", code: "COLOR_OFF_BRAND", message: "Non-brand shape color detected.", pageId: page.id, elementId: el.id });
        }
      }
    }
  }

  // Required logo/footer if strict:
  if (!brand.allowOffBrand) {
    const hasLogo = pages.some(p => p.elements.some(e => e.tag === "logo"));
    if (!hasLogo) issues.push({ severity: "error", code: "MISSING_LOGO", message: "Logo is required in strict brand mode." });

    const hasFooter = pages.some(p => p.elements.some(e => e.tag === "footer"));
    if (!hasFooter) issues.push({ severity: "error", code: "MISSING_FOOTER", message: "Footer is required in strict brand mode." });
  }

  return issues;
}
D) Gate runner (single entry point)
async function runPrePublishGate(params: {
  pages: CanvasPage[];
  brand: BrandKit;
  metadata: { title?: string; examTarget?: string; locale?: string; level?: string; };
  measureText: (t: Extract<CanvasElement,{type:"text"}>) => { overflow: boolean };
  requireStoreReadyAssets?: boolean;
  hasStoreReadyAssets?: boolean;
  clinicalReview?: boolean;
}): Promise<{ ok: boolean; issues: GateIssue[] }> {
  const issues: GateIssue[] = [];

  // metadata
  if (!params.metadata.title) issues.push({ severity: "error", code: "MISSING_TITLE", message: "Title is required." });
  if (!params.metadata.examTarget) issues.push({ severity: "error", code: "MISSING_EXAM", message: "Exam target is required." });
  if (!params.metadata.locale) issues.push({ severity: "error", code: "MISSING_LOCALE", message: "Locale is required." });
  if (!params.metadata.level) issues.push({ severity: "error", code: "MISSING_LEVEL", message: "Level is required." });

  // overflow
  issues.push(...scanOverflow(params.pages, params.measureText));

  // brand
  issues.push(...scanBrandCompliance(params.pages, params.brand));

  // store assets requirement
  if (params.requireStoreReadyAssets && !params.hasStoreReadyAssets) {
    issues.push({ severity: "error", code: "MISSING_MEDIA_PACK", message: "Store-ready media pack has not been generated." });
  }

  // optional clinical review (soft -> warn or hard -> error depending on your preference)
  if (params.clinicalReview) {
    const text = extractStudyTextFromCanvas(params.pages);
    const res = await fetch("/api/ai/clinical-review", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ text, examTarget: params.metadata.examTarget, locale: params.metadata.locale }),
    });
    if (res.ok) {
      const out = await res.json() as { flags: { severity:"warn"|"error"; message:string }[] };
      for (const f of out.flags ?? []) {
        issues.push({ severity: f.severity, code: "CLINICAL_FLAG", message: f.message });
      }
    } else {
      issues.push({ severity: "warn", code: "CLINICAL_REVIEW_FAILED", message: "Clinical review could not run. You can retry or publish without it." });
    }
  }

  const ok = !issues.some(i => i.severity === "error");
  return { ok, issues };
}

4) The “Store-Ready Pipeline” button (orchestrator)

This should do exactly what you want: one button → clean → paginate → autofit → generate assets → pass gate.

async function runStoreReadyPipeline(params: {
  pages: CanvasPage[];
  brand: BrandKit;
  // your existing helpers:
  applyAutofitToPage: (pageId?: string) => Promise<void>;
  fixOverflowWithPagination: () => Promise<void>; // your new multi-page AI pagination + continuation logic
  enforceBrandOnExport: () => Promise<void>; // normalize fonts/colors/logo/footer
  generateMediaPack: () => Promise<MediaPack>;
  saveMediaPack: (pack: MediaPack) => Promise<void>;
  gate: () => Promise<{ ok: boolean; issues: GateIssue[] }>;
}): Promise<{ ok: boolean; issues?: GateIssue[] }> {
  // 1) Autofit everything
  await params.applyAutofitToPage(undefined);

  // 2) Multi-page pagination/fix overflow
  await params.fixOverflowWithPagination();

  // 3) Enforce brand at export time (even if allowOffBrand during design)
  await params.enforceBrandOnExport();

  // 4) Generate + persist listing assets
  const pack = await params.generateMediaPack();
  await params.saveMediaPack(pack);

  // 5) Gate check
  const g = await params.gate();
  return g.ok ? { ok: true } : { ok: false, issues: g.issues };
}
