// src/lib/admin-fetch.ts
type AdminCreds = { username: string; password: string };

function getAdminCreds(): AdminCreds | null {
  // Adjust these to match wherever you store admin creds.
  // Common patterns in Replit apps:
  // - localStorage set during admin login
  // - Vite env variables (not recommended for secrets)
  // - a dedicated admin login modal that stores them client-side
  const username =
    (typeof window !== "undefined" && localStorage.getItem("ADMIN_USERNAME")) || "";
  const password =
    (typeof window !== "undefined" && localStorage.getItem("ADMIN_PASSWORD")) || "";

  if (!username || !password) return null;
  return { username, password };
}

/**
 * Adds admin credentials to a URL query string.
 * Used for GET/HEAD because requireAdmin checks req.query there.
 */
export function getAdminParams(): string {
  const creds = getAdminCreds();
  if (!creds) return "";
  const sp = new URLSearchParams();
  sp.set("username", creds.username);
  sp.set("password", creds.password);
  return sp.toString();
}

/**
 * Unified admin fetch:
 * - GET/HEAD: inject creds into query params
 * - non-GET: inject creds into JSON body
 *
 * Contract: returns fetch Response (so callers can use res.ok/res.json()).
 */
export async function adminFetch(
  url: string,
  init: (RequestInit & { body?: any }) = {}
): Promise<Response> {
  const method = (init.method || "GET").toUpperCase();
  const creds = getAdminCreds();

  // Normalize URL to allow relative paths
  const isAbsolute = /^https?:\/\//i.test(url);
  const base = (typeof window !== "undefined" ? window.location.origin : "http://localhost");
  const u = new URL(url, isAbsolute ? undefined : base);

  const headers = new Headers(init.headers || {});
  const out: RequestInit = { ...init, method, headers };

  // Ensure JSON header for non-GET when body is present
  const ensureJsonHeader = () => {
    if (!headers.has("Content-Type")) headers.set("Content-Type", "application/json");
  };

  if (method === "GET" || method === "HEAD") {
    // ✅ requireAdmin checks req.query for GET/HEAD
    if (creds) {
      u.searchParams.set("username", creds.username);
      u.searchParams.set("password", creds.password);
    }
    // Never send a body on GET/HEAD
    delete (out as any).body;
  } else {
    // ✅ requireAdmin checks req.body for non-GET
    // Merge creds into body (object) or wrap if body is empty
    let bodyObj: any = {};

    if (init.body !== undefined && init.body !== null) {
      if (typeof init.body === "string") {
        try {
          bodyObj = JSON.parse(init.body);
        } catch {
          // If caller passed a raw string (rare), send as-is but still include creds
          bodyObj = { payload: init.body };
        }
      } else if (typeof init.body === "object") {
        bodyObj = init.body;
      } else {
        bodyObj = { payload: init.body };
      }
    }

    if (creds) {
      // Only set if not already provided by caller
      if (bodyObj.username === undefined) bodyObj.username = creds.username;
      if (bodyObj.password === undefined) bodyObj.password = creds.password;
    }

    ensureJsonHeader();
    out.body = JSON.stringify(bodyObj);
  }

  // For relative urls, use pathname+search to avoid leaking "localhost" base in SSR
  const finalUrl = isAbsolute ? u.toString() : `${u.pathname}${u.search}`;

  return fetch(finalUrl, out);
}
2) Fix your call sites (required in your snippet)

Once the above is in, you should stop building query strings manually and stop using raw fetch for admin routes.

Replace these patterns:

✅ GET

fetch(`/api/admin/design-projects?${getAdminParams()}`)

➡️

adminFetch(`/api/admin/design-projects`)

✅ DELETE

fetch(`/api/admin/design-pages/${pageId}?${getAdminParams()}`, { method: "DELETE" })

➡️

adminFetch(`/api/admin/design-pages/${pageId}`, { method: "DELETE" })

✅ POST/PUT stop pre-stringifying bodies:

adminFetch("/api/admin/design-pages/123", {
  method: "PUT",
  body: JSON.stringify({ canvasJson: {...} })
})

➡️

adminFetch("/api/admin/design-pages/123", {
  method: "PUT",
  body: { canvasJson: {...} }
})
3) Security note (important)

This model inherently puts admin creds into the URL for GET requests, which can leak into:

server logs

reverse proxy logs

browser history

analytics

If you can change requireAdmin, the best upgrade is:

accept creds from headers (Authorization: Basic … or X-Admin-Token)

or switch to cookie/session

But given your current constraint, the code above at least ensures:

query creds only on GET/HEAD (where required)

body creds on mutations (no URL leaks there)
