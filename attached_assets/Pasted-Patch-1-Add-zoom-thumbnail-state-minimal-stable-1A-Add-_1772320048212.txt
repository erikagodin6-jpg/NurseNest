1C) Add makeThumb helper (put it below renderPageToCanvas)

Directly under your renderPageToCanvas function, paste:

const makeThumb = useCallback((pageObjects: CanvasObject[], bgColor: string) => {
  const src = renderPageToCanvas(pageObjects, bgColor);

  const targetW = 160;
  const targetH = 210;

  const t = document.createElement("canvas");
  t.width = targetW * 2;
  t.height = targetH * 2;

  const ctx = t.getContext("2d")!;
  ctx.fillStyle = bgColor;
  ctx.fillRect(0, 0, t.width, t.height);

  const scale = Math.min(t.width / src.width, t.height / src.height);
  const dx = (t.width - src.width * scale) / 2;
  const dy = (t.height - src.height * scale) / 2;

  ctx.drawImage(src, dx, dy, src.width * scale, src.height * scale);
  return t.toDataURL("image/png");
}, [renderPageToCanvas]);
