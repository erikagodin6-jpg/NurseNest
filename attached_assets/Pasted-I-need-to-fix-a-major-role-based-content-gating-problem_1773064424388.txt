function getVisibleLessons(lessons, user) {
  if (!user) {
    return lessons.filter(lesson =>
      lesson.lessonTier === "free" || lesson.isFreePreview === true
    );
  }

  if (user.role === "free") {
    return lessons.filter(lesson =>
      lesson.lessonTier === "free" || lesson.isFreePreview === true
    );
  }

  if (user.role === "rpn") {
    return lessons.filter(lesson =>
      lesson.lessonTier === "rpn" || lesson.lessonTier === "free"
    );
  }

  if (user.role === "rn") {
    return lessons.filter(lesson =>
      lesson.lessonTier === "rn" || lesson.lessonTier === "free"
    );
  }

  if (user.role === "np") {
    return lessons.filter(lesson =>
      lesson.lessonTier === "np" || lesson.lessonTier === "free"
    );
  }

  if (user.role === "admin") {
    return lessons;
  }

  return [];
}
