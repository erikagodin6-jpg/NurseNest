    function renderLessonCards() {
      const container = document.getElementById('lessons-grid');
      const lessons = interactiveLessons[userSubscriptionTier] || [];
      
      container.innerHTML = lessons.map(lesson => `

          <div class="w-12 h-12 bg-${lesson.color}-100 rounded-lg flex items-center justify-center text-2xl mb-4">
            ${lesson.system === 'Cardiovascular' ? '❤️' : lesson.system === 'Respiratory' ? '🫁' : lesson.system === 'Neurological' ? '🧠' : lesson.system === 'Infection Control' ? '🔬' : '💊'}

    function showLesson(lessonId) {
      const lesson = interactiveLessons[userSubscriptionTier].find(l => l.id === lessonId);
      if (!lesson) return;

      document.getElementById('active-lesson-title').textContent = lesson.title;
      document.getElementById('active-lesson-system').textContent = lesson.system;
      document.getElementById('lesson-scenario').textContent = lesson.scenario;

      document.getElementById('lesson-symptoms').innerHTML = lesson.symptoms
        .map(s => `<div class="flex items-start gap-2 p-2 bg-gray-50 rounded"><span>${s}</span></div>`)
        .join('');

      document.getElementById('lesson-assessment').innerHTML = lesson.assessment
        .map(a => `<div class="flex items-start gap-2 p-2 bg-gray-50 rounded"><span class="text-lavender-500 mr-2">✓</span><span>${a}</span></div>`)
        .join('');

      document.getElementById('lesson-interventions').innerHTML = lesson.interventions
        .map(i => `<div class="flex items-start gap-2 p-2 bg-green-50 rounded"><span class="text-mint-500 mr-2">→</span><span>${i}</span></div>`)
        .join('');

      document.getElementById('quiz-container').innerHTML = `
        <p class="text-gray-800 font-medium mb-4">${lesson.quiz.question}</p>
        <div class="space-y-2">
          ${lesson.quiz.options.map((opt, idx) => `
            <button class="quiz-option w-full text-left p-3 border-2 border-gray-200 rounded-lg hover:border-lavender-300 transition" data-idx="${idx}">
              ${opt}
            </button>
          `).join('')}
        </div>
      `;

      document.querySelectorAll('.quiz-option').forEach(opt => {
        opt.addEventListener('click', () => {
          const isCorrect = parseInt(opt.dataset.idx) === lesson.quiz.correct;
          opt.classList.add(isCorrect ? 'border-mint-500 bg-mint-50' : 'border-red-500 bg-red-50');
          opt.disabled = true;
        });
      });

      document.getElementById('lesson-panel').classList.remove('hidden');
    }

    function mapToCapabilities(config) {
      return {
        recolorables: [
          {
            get: () => config.background_color || defaultConfig.background_color,
            set: (value) => {
              config.background_color = value;
              window.elementSdk.setConfig({ background_color: value });
            }
          },
          {
            get: () => config.primary_action_color || defaultConfig.primary_action_color,
            set: (value) => {
              config.primary_action_color = value;
              window.elementSdk.setConfig({ primary_action_color: value });
            }
          }
        ],
        borderables: [],
        fontEditable: {
          get: () => config.font_family || defaultConfig.font_family,
          set: (value) => {
            config.font_family = value;
            window.elementSdk.setConfig({ font_family: value });
          }
        },
        fontSizeable: {
          get: () => config.font_size || defaultConfig.font_size,
          set: (value) => {
            config.font_size = value;
            window.elementSdk.setConfig({ font_size: value });
          }
        }
      };
    }

    function applyTheme(themeName) {
      const theme = themes[themeName];
      const root = document.documentElement;
      
      // Update CSS variables
      root.style.setProperty('--primary-color', theme.primary);
      root.style.setProperty('--secondary-color', theme.secondary);

      if (logoText) {

      if (nav) {
        nav.style.borderBottomColor = theme.primary + '20';
      }

      // Update all theme-dependent colors in the interface
      updateThemeColors(theme);
      
      // Save preference
      localStorage.setItem('nursennest-theme', themeName);
    }

    function updateThemeColors(theme) {
      // Remove any existing theme style
      const existingStyle = document.getElementById('theme-style');
      if (existingStyle) {
        existingStyle.remove();
      }
      
      // Create new theme style
      const root = document.documentElement;
      const style = document.createElement('style');
      style.id = 'theme-style';
      style.textContent = `
        :root {
          --primary: ${theme.primary};
          --secondary: ${theme.secondary};
          --tertiary: ${theme.tertiary};
          --quaternary: ${theme.quaternary};
        }
      `;
      document.head.appendChild(style);
      
      // Update tier tabs color
      document.querySelectorAll('.tier-tab.text-lavender-500').forEach(tab => {
        tab.style.color = theme.primary;
      });
      
      document.querySelectorAll('.tier-tab.border-lavender-500').forEach(tab => {
        tab.style.borderBottomColor = theme.primary;
      });
    }

    function renderTierLessons(tier) {
      const contentGrid = document.getElementById('content-grid');
      const tierTitle = document.getElementById('tier-title');
      const tierDesc = document.getElementById('tier-description');
      const paywallView = document.getElementById('paywall-view');
      const tierContentView = document.getElementById('tier-content');
      
      paywallView.classList.add('hidden');
      tierContentView.classList.remove('hidden');

      // Update title and description based on tier
      const tierNames = {
        rpn: { title: 'RPN Essentials', description: 'Foundational nursing practice and common patient presentations' },
        rn: { title: 'RN Critical Care', description: 'Advanced assessment and management of critically ill patients' },
        np: { title: 'NP Advanced Practice', description: 'Complex decision-making and advanced clinical interventions' }
      };
      const tierInfo = tierNames[tier] || tierNames.rn;
      tierTitle.textContent = tierInfo.title;
      tierDesc.textContent = tierInfo.description;

      const lessons = systemLessons[tier] || [];
      contentGrid.innerHTML = '';
      
      lessons.forEach((lessonItem, idx) => {
        const bgClass = `${lessonItem.color}-50`;
        const borderClass = `border-${lessonItem.color}-200`;
        const textClass = lessonItem.color === 'lavender' ? 'text-lavender-600' :
                         lessonItem.color === 'blush' ? 'text-blush-500' :
                         lessonItem.color === 'powder' ? 'text-powder-500' :
                         'text-mint-500';
        
        const sysCard = document.createElement('div');
        sysCard.className = `bg-white rounded-2xl p-6 shadow-sm hover:shadow-lg transition cursor-pointer border-2 ${borderClass}`;
        sysCard.dataset.lessonIdx = idx;
        
        sysCard.innerHTML = `
          <div class="flex items-start gap-3 mb-4">
            <div class="text-3xl">${lessonItem.system.split(' ')[0]}</div>
            <div class="flex-1">
              <h3 class="text-lg font-semibold text-gray-800">${lessonItem.system}</h3>
              <p class="text-xs text-softgray">${lessonItem.scenarios.length} clinical scenarios</p>
            </div>
          </div>
          <div class="bg-${bgClass} rounded-lg p-3 mb-4 border-l-4 border-${lessonItem.color}-400">
            <p class="text-xs text-gray-700 line-clamp-3">${lessonItem.pathophysiology}</p>
          </div>
          <button class="w-full py-2 px-3 bg-${bgClass} ${textClass} font-medium rounded-lg hover:opacity-90 transition text-sm">
            Explore Scenarios →
          </button>
        `;
        
        sysCard.addEventListener('click', () => {
          showLessonDetail(tier, idx);
        });
        
        contentGrid.appendChild(sysCard);
      });
    }
