B) Generator deep-link preselect via query param
In your generator page:
import { useLocation } from "wouter";

function getQueryType(search: string) {
  const params = new URLSearchParams(search || "");
  return params.get("type"); // bundle / cram / question-pack / cheat-sheet
}

export function ProductGeneratorPage() {
  const [location] = useLocation();
  const search = location.includes("?") ? location.split("?")[1] : "";
  const type = getQueryType(search);

  // on mount, preselect template
  useEffect(() => {
    if (!type) return;
    setSelectedTemplate(type);
  }, [type]);

  // ...
}
