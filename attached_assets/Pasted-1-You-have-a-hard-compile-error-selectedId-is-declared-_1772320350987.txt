const mapColorToActiveTheme = (color?: string) => {

Then replace applyThemeToObjects usage for the “all pages” case:

const applyActiveThemeToObjects = (objs: CanvasObject[]): CanvasObject[] => {
  return objs.map(obj => {
    const updated = { ...obj };
    updated.fill = mapColorToActiveTheme(obj.fill);
    updated.stroke = mapColorToActiveTheme(obj.stroke);
    if (obj.type === "text") {
      const isHeading = (obj.fontSize || 0) >= 18 || obj.fontWeight === "bold";
      updated.fontFamily = isHeading ? theme.headingFont : theme.bodyFont;
    }
    return updated;
  });
};

And in applyThemeToAllPages:

3) 🖼️ You declared pageThumbs but never generate thumbnails

You already have renderPageToCanvas(). Perfect. You just need a tiny “makeThumb” + effect.

✅ Add these helpers below renderPageToCanvas:
const makeThumb = useCallback((pageObjects: CanvasObject[], bg: string) => {
  const c = renderPageToCanvas(pageObjects, bg);
  // Small thumbnail canvas
  const tw = 140;
  const th = 180;
  const out = document.createElement("canvas");
  out.width = tw;
  out.height = th;
  const ctx = out.getContext("2d")!;
  ctx.fillStyle = bg;
  ctx.fillRect(0, 0, tw, th);

  const scale = Math.min(tw / c.width, th / c.height);
  const dx = (tw - c.width * scale) / 2;
  const dy = (th - c.height * scale) / 2;
  ctx.drawImage(c, dx, dy, c.width * scale, c.height * scale);

  return out.toDataURL("image/png");
}, [renderPageToCanvas]);
