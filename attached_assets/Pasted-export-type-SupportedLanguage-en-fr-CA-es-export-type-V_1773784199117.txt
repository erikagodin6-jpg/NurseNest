export type SupportedLanguage = "en" | "fr-CA" | "es";

export type ValidationStatus =
  | "approved"
  | "validation_failed"
  | "needs_review";

export type FieldValidationResult = {
  field: string;
  requestedLanguage: SupportedLanguage;
  detectedLanguage: string;
  passed: boolean;
  reason?: string;
};

export type GeneratorValidationResult = {
  requestedLanguage: SupportedLanguage;
  detectedLanguageOverall: string;
  fieldValidation: FieldValidationResult[];
  validationPassed: boolean;
  terminologyCheckPassed: boolean;
  retryCount: number;
  status: ValidationStatus;
};

export type GeneratedEducationalPayload = {
  title?: string;
  stem?: string;
  scenario?: string;
  answerChoices?: string[];
  correctRationale?: string;
  incorrectRationales?: string[];
  explanation?: string;
  clinicalPearls?: string[];
  hints?: string[];
  lessonSections?: Array<{ heading: string; body: string }>;
  metadataLabels?: string[];
};

const LANGUAGE_HINTS: Record<SupportedLanguage, RegExp[]> = {
  en: [
    /\bthe\b/i,
    /\band\b/i,
    /\bpatient\b/i,
    /\bnurse\b/i,
    /\bwhich\b/i,
    /\bselect all that apply\b/i,
  ],
  "fr-CA": [
    /\ble\b/i,
    /\bla\b/i,
    /\bles\b/i,
    /\bet\b/i,
    /\bpatient\b/i,
    /\binfirmi(ère|er|ères|ers)\b/i,
    /\bsélectionnez\b/i,
    /\bréponse\b/i,
  ],
  es: [
    /\bel\b/i,
    /\bla\b/i,
    /\blos\b/i,
    /\by\b/i,
    /\bpaciente\b/i,
    /\benfermer[oa]s?\b/i,
    /\bseleccione\b/i,
    /\brespuesta\b/i,
  ],
};

const TERMINOLOGY_DICTIONARY: Record<SupportedLanguage, string[]> = {
  en: [
    "active recall",
    "spaced repetition",
    "clinical judgment",
    "nursing interventions",
    "patient education",
    "lab values",
  ],
  "fr-CA": [
    "rappel actif",
    "répétition espacée",
    "jugement clinique",
    "interventions infirmières",
    "enseignement au patient",
    "valeurs de laboratoire",
  ],
  es: [
    "recuperación activa",
    "repetición espaciada",
    "juicio clínico",
    "intervenciones de enfermería",
    "educación al paciente",
    "valores de laboratorio",
  ],
};

function normalizeForDetection(text: string): string {
  return text.trim().replace(/\s+/g, " ");
}

export function detectLanguage(text: string): string {
  const normalized = normalizeForDetection(text);
  if (!normalized) return "unknown";

  const scores = Object.entries(LANGUAGE_HINTS).map(([lang, patterns]) => {
    let score = 0;
    for (const rx of patterns) {
      if (rx.test(normalized)) score += 1;
    }
    return { lang, score };
  });

  scores.sort((a, b) => b.score - a.score);
  if (scores[0].score === 0) return "unknown";
  return scores[0].lang;
}

function flattenPayload(payload: GeneratedEducationalPayload): Array<{ field: string; value: string }> {
  const rows: Array<{ field: string; value: string }> = [];

  if (payload.title) rows.push({ field: "title", value: payload.title });
  if (payload.stem) rows.push({ field: "stem", value: payload.stem });
  if (payload.scenario) rows.push({ field: "scenario", value: payload.scenario });
  if (payload.correctRationale) rows.push({ field: "correctRationale", value: payload.correctRationale });
  if (payload.explanation) rows.push({ field: "explanation", value: payload.explanation });

  payload.answerChoices?.forEach((value, i) => rows.push({ field: `answerChoices[${i}]`, value }));
  payload.incorrectRationales?.forEach((value, i) => rows.push({ field: `incorrectRationales[${i}]`, value }));
  payload.clinicalPearls?.forEach((value, i) => rows.push({ field: `clinicalPearls[${i}]`, value }));
  payload.hints?.forEach((value, i) => rows.push({ field: `hints[${i}]`, value }));
  payload.metadataLabels?.forEach((value, i) => rows.push({ field: `metadataLabels[${i}]`, value }));

  payload.lessonSections?.forEach((section, i) => {
    rows.push({ field: `lessonSections[${i}].heading`, value: section.heading });
    rows.push({ field: `lessonSections[${i}].body`, value: section.body });
  });

  return rows;
}

export function checkTerminology(
  payload: GeneratedEducationalPayload,
  requestedLanguage: SupportedLanguage
): boolean {
  const flattened = flattenPayload(payload)
    .map((r) => r.value.toLowerCase())
    .join(" ");

  const forbiddenTerms = Object.entries(TERMINOLOGY_DICTIONARY)
    .filter(([lang]) => lang !== requestedLanguage)
    .flatMap(([, terms]) => terms);

  return !forbiddenTerms.some((term) => flattened.includes(term.toLowerCase()));
}

export function validateGeneratedPayload(
  payload: GeneratedEducationalPayload,
  requestedLanguage: SupportedLanguage,
  retryCount = 0
): GeneratorValidationResult {
  const rows = flattenPayload(payload);

  const fieldValidation: FieldValidationResult[] = rows.map((row) => {
    const detected = detectLanguage(row.value);
    const passed = detected === requestedLanguage || detected === "unknown";

    return {
      field: row.field,
      requestedLanguage,
      detectedLanguage: detected,
      passed,
      reason: passed ? undefined : `Detected ${detected}, expected ${requestedLanguage}`,
    };
  });

  const overallText = rows.map((r) => r.value).join("\n");
  const detectedLanguageOverall = detectLanguage(overallText);
  const terminologyCheckPassed = checkTerminology(payload, requestedLanguage);
  const validationPassed =
    fieldValidation.every((f) => f.passed) &&
    (detectedLanguageOverall === requestedLanguage || detectedLanguageOverall === "unknown") &&
    terminologyCheckPassed;

  return {
    requestedLanguage,
    detectedLanguageOverall,
    fieldValidation,
    validationPassed,
    terminologyCheckPassed,
    retryCount,
    status: validationPassed ? "approved" : "validation_failed",
  };
}
