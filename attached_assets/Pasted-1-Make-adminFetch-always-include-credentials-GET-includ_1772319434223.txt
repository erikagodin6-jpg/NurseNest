1) Make adminFetch always include credentials (GET included)

You wrote: “lib version only adds credentials for non-GET methods.”
That will break all admin GET routes once you remove getAdminParams().

Fix src/lib/admin-fetch.ts so credentials are always included.

If you want to keep your current “returns Response” contract, use this shape:

// src/lib/admin-fetch.ts
export function adminFetch(url: string, init: RequestInit & { body?: any } = {}) {
  const method = (init.method || "GET").toUpperCase();

  const headers = new Headers(init.headers || {});
  const out: RequestInit = {
    ...init,
    method,
    credentials: "include", // ✅ ALWAYS
    headers,
  };

  // ✅ Never send body on GET/HEAD
  if (method === "GET" || method === "HEAD") {
    delete (out as any).body;
  } else if (init.body !== undefined) {
    // allow callers to pass object OR string
    if (typeof init.body === "string") {
      out.body = init.body;
      if (!headers.has("Content-Type")) headers.set("Content-Type", "application/json");
    } else {
      out.body = JSON.stringify(init.body);
      headers.set("Content-Type", "application/json");
    }
  }

  return fetch(url, out); // ✅ returns Response
}

// getAdminParams should become a no-op or be deleted
export function getAdminParams() {
  return ""; // or remove entirely
}

Outcome: you can remove all ?${getAdminParams()} and admin GETs will still authenticate via cookie/session.
